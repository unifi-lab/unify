## Server Setup

### 1. Install Dependencies

```bash
npm install @unilab/unify-hono @unilab/unify-client @unilab/core
```

### 2. Create Server

```typescript
// server.ts
import { Unify } from "@unilab/unify-hono";
import { WalletPlugin } from "@unilab/uniweb3";
import { Logging } from "@unilab/core/middleware";

const app = Unify.init({
  plugins: [WalletPlugin],
  middleware: [Logging()],
});

export default {
  port: 3000,
  fetch: app.fetch,
};
```

### 3. Client Usage

```typescript
// client.ts
import { repo, UnifyClient } from "@unilab/unify-client";
import { WalletEntity } from "@unilab/uniweb3/entities";

UnifyClient.init({
  baseUrl: "http://localhost:3000",
  timeout: 10000,
});

const evmBalanceRes = await repo<WalletEntity>({
  entityName: "wallet",
  source: "evm",
}).findOne({
  where: {
    address: "0x...",
    network: "ethereum",
  },
});

const solanaBalanceRes = await repo<WalletEntity>({
  entityName: "wallet",
  source: "solana",
}).findOne({
  where: {
    address: "11111111111111111111111111111112",
  },
});
```

## Running the Server

```bash
bun run server.ts
# or
node server.js
```
